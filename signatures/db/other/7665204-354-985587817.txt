

    public int readByte() {
        if ( bytes_in_buf == -1 ) throw new InputMismatchException();
        if ( cur_byte_idx >= bytes_in_buf ) {
            cur_byte_idx = 0;
            try {
                bytes_in_buf = is.read( buf );
            } catch ( IOException e ) {
                throw new InputMismatchException();
            }
            if ( bytes_in_buf <= 0 ) return -1;
        }
        return buf[cur_byte_idx ++];
    }
